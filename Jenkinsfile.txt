pipeline {
    agent any
         stage('SCM checkout'){
             steps{
            git 'https://github.com/shawla4real/Module-terraform.git'
         }
    }
        stage('Terraform Init'){
         steps{
             sh '''terraform init
''' 
         }   
        }
        stage('Terraform plan'){
         steps{
             sh '''terraform plan
''' 
         }   
        }
        stage('Terraform apply'){
         steps{
             sh '''terraform apply --auto-approve
'''  
         }   
        }
    }
